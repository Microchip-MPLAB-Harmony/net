TCPIP_FTPC_STATUS_TYPE Structure

Structure: TCPIP_FTPC_STATUS_TYPE 

This data structure describes the details of FTPC errors.

C
typedef struct {
  NET_PRES_SKT_HANDLE_T ctrlSocket;
  NET_PRES_SKT_HANDLE_T dataSocket;
  TCPIP_FTPC_CMD cmd;
  TCPIP_FTPC_ERROR_TYPE error;
  uint32_t dataRxLen;
  uint32_t dataTxLen;
  uint16_t ctrlRxLen;
  uint16_t ctrlTxLen;
  bool busy;
  bool isConnected;
  bool isLoggedIn;
} TCPIP_FTPC_STATUS_TYPE;
