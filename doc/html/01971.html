TCPIP_DHCPV6_IA_INFO Structure

DHCPv6 IA info

C
typedef struct {
  TCPIP_DHCPV6_IA_TYPE iaType;
  TCPIP_DHCPV6_IA_STATE iaState;
  TCPIP_DHCPV6_IA_SUBSTATE iaSubState;
  int iaIndex;
  uint32_t iaId;
  uint32_t tAcquire;
  uint32_t t1;
  uint32_t t2;
  IPV6_ADDR ipv6Addr;
  uint32_t prefLTime;
  uint32_t validLTime;
  TCPIP_DHCPV6_SERVER_STATUS_CODE lastStatusCode;
  void* statusBuff;
  size_t statusBuffSize;
} TCPIP_DHCPV6_IA_INFO;

IA state

preferred life time for the IPv6 address; seconds 
uint32_t validLTime; 
valid life time for the IPv6 address; seconds 
TCPIP_DHCPV6_SERVER_STATUS_CODE lastStatusCode; 
last status code for this IA 
void* statusBuff; 
buffer to copy the latest status message associated with this IA

size of this buffer
